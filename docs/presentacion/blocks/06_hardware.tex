\begin{frame}{Evolución del Controlador: Puente Serie ESP32}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{alertblock}{Fase 1 — BLE directo (heredado)}
      \footnotesize
      \begin{itemize}
        \item \texttt{fanlamp\_bt.py} $\rightarrow$ \texttt{btmgmt} (\texttt{bluez})
        \item Requiere \texttt{sudo} (parar/reiniciar \texttt{bluetoothd})
        \item Monopoliza el adaptador BT del servidor
        \item $\sim$3 s por comando, inestable
      \end{itemize}
    \end{alertblock}

    \column{0.52\textwidth}
    \begin{exampleblock}{Fase 2 — Puente ESP32 (actual)}
      \footnotesize
      \begin{itemize}
        \item \texttt{fanlamp\_control.py} $\rightarrow$ USB serie (115200 baud)
        \item ESP32 como radio BLE externa dedicada
        \item Sin \texttt{sudo} (grupo \texttt{dialout})
        \item $\sim$1,5 s por comando, reset DTR automático
        \item Adaptador BT del servidor libre para Xiaomi
      \end{itemize}
    \end{exampleblock}
  \end{columns}

  \begin{block}{Ventaja clave}
    \footnotesize
    \textbf{PC $\rightarrow$ USB serie $\rightarrow$ ESP32 $\rightarrow$ BLE $\rightarrow$ FanLamp}\\[0.1cm]
    El ESP32 que hoy es puente BLE podrá ser puente LIN para la HVAC.
    \textbf{Hardware reutilizado, coste 0 € adicional.}
    \vspace{0.15cm}
  \end{block}
\end{frame}
