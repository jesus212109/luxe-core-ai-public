\begin{frame}{3 — Restricciones y Arquitectura Local}
  \hypertarget{s:restricciones}{}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \textbf{Restricciones de partida}
    \begin{itemize}
      \item Un curso académico $\rightarrow$ MVP funcional
      \item Presupuesto: \textbf{56,50 €} — hardware de consumo
      \item Ingeniería inversa como alternativa a la inversión
    \end{itemize}

    \bigskip
    \textbf{Infraestructura}
    \begin{itemize}
      \item Fase 1: servidores UCO con GPU (prototipado)
      \item Fase 2: torre Ryzen 5, 16 GB RAM, 100\% local
    \end{itemize}

    \column{0.5\textwidth}
    \footnotesize
    \textbf{Arquitectura de red air-gapped}
    \begin{itemize}
      \item Router TL-MR6400 sin tarjeta SIM — sin salida WAN
      \item Segmento \texttt{192.168.1.0/24} aislado, IPs estáticas
      \item Tráfico no puede abandonar la subred
      \item Verificable: retirar la SIM demuestra el diseño
    \end{itemize}

    \begin{alertblock}{Principio clave}
      \centering\footnotesize
      Domótica inteligente $\neq$ grandes inversiones.
      La interoperabilidad no depende de la nube: depende del software.
    \end{alertblock}
  \end{columns}
\end{frame}
