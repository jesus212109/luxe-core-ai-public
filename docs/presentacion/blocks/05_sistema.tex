\begin{frame}{Asesores de Confort Térmico}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \footnotesize
    \textbf{Comfort Advisor (ASHRAE 55)}
    \begin{itemize}
      \item Índice de calor NOAA (Rothfusz)
      \item 6 zonas de confort adaptadas al clima andaluz
      \item Evaluación de humedad (seco / óptimo / húmedo)
      \item Detección de tendencia ($^\circ$C/hora)
    \end{itemize}

    \bigskip
    \textbf{Zonas térmicas:}
    \begin{enumerate}
      \scriptsize
      \item Fresco ($<$ 23$^\circ$C) — sin acción
      \item Confortable (23--26$^\circ$C)
      \item Cálido (26--28$^\circ$C) — ventilador 1
      \item Caluroso (28--31$^\circ$C) — ventilador 3
      \item Muy caluroso (31--34$^\circ$C) — ventilador 4
      \item Extremo ($>$ 34$^\circ$C) — ventilador 5, AC urgente
    \end{enumerate}

    \column{0.5\textwidth}
    \footnotesize
    \textbf{Smart Advisor (contexto exterior)}
    \begin{itemize}
      \item Comparación interior vs.\ exterior (\texttt{wttr.in})
      \item Lógica ventana/persiana (dif. térmica, UV, lluvia)
      \item Detección de presencia (manual/BLE/IP Wi-Fi)
      \item \textbf{\textit{Human in the Loop}:} preguntar antes de actuar
    \end{itemize}

    \bigskip
    \textbf{Auto-acciones condicionales:}
    \begin{itemize}
      \scriptsize
      \item Misma zona $\rightarrow$ no actuar
      \item +1 zona $\rightarrow$ preguntar
      \item +2 zonas $\rightarrow$ actuar automático
      \item Zona extrema $\rightarrow$ actuar siempre
      \item Sin presencia $\rightarrow$ nunca actuar
    \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Auto-Recuperación y Tolerancia a Fallos}

      \textcolor{darkgray}{[Figura: ciclo de auto-recuperación]}

    \column{0.48\textwidth}
    \begin{exampleblock}{Flujo de recuperación}
      \scriptsize
      \textbf{SelfHealManager} verifica cada 30\,s la salud de
      Ollama, Circuit Breaker y ESP32. Ante un fallo:
      \begin{enumerate}
        \item \textbf{Circuit Breaker} aísla el servicio
              (\texttt{CLOSED} $\to$ \texttt{OPEN})
        \item \textbf{Reintentos} con backoff + jitter (1\,s, 3\,s)
        \item Si persiste: \textbf{reset USB} del ESP32
      \end{enumerate}
      Sin intervención humana.
    \end{exampleblock}
  \end{columns}
\end{frame}
